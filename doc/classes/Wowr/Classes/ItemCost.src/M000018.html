                        def initialize(elem)
                                @buy_price  = elem[:buyPrice].to_i  if elem[:buyPrice]
                                @sell_price = elem[:sellPrice].to_i if elem[:sellPrice]
                                
                                if (elem%'token')
                                        @tokens = []
                                        (elem/:token).each do |token|
                                                @tokens << ItemCostToken.new(token)
                                        end
                                end
                        end
