                        def initialize(elem)
                                @name = elem[:name]
                                @id = elem[:id].to_i
                                @icon = elem[:icon]
                                
                                # not all items have reagents, some are just spells
                                if (elem%'reagent')
                                        @reagents = []
                                        (elem/:reagent).each do |reagent|
                                                @reagents << Reagent.new(reagent)
                                        end
                                end
                        end
