                        def initialize(elem)
                                @name = elem[:name]
                                
                                @items = []
                                (elem/:item).each do |item|
                                        @items << item[:name]
                                end
                                
                                @set_bonuses = []
                                (elem/:setBonus).each do |bonus|
                                        @set_bonuses << ItemSetBonus.new(bonus)
                                end
                        end
