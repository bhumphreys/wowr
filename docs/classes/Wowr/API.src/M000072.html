                def get_item_info(options = {:item_id => nil, :locale => @locale})
                        xml = get_xml(@@item_info_url, options)
                        if (xml%'itemInfo'%'item')
                                return Wowr::Classes::ItemInfo.new(xml%'itemInfo'%'item')
                        else
                                return nil
                        end
                end
